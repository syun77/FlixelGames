// ザコ敵をばらまく
$10 = 4
while($10 > 0) {
	ENEMY2(3, RED, RND(120)-60, 150)
	WAIT(100)
	$10 -= 1
}

WAIT(500)
$11 = 2
while($11 > 0) {
	// 跳ね返る敵をばらまく
	$10 = 8
	$1 = 100 + (10 * RANK())
	while($10 > 0) {
		ENEMY2(9, BLUE, 120+RND(120), $1)
		$10 -= 1
	}
	WAIT(2500)
	$11 -= 1
}
