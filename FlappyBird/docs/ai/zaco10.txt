// ================================
// ザコ10 - 左右に動く。Whip弾を撃つ
// ================================
// ログ有効
//LOG(1)
DECAY(93)
WAIT(2000)

// 左右に動く
REFLECT()
DECAY(100)
if(LOT()) {
	MOVE(90, 80)
}
else {
	MOVE(270, 80)
}
while(1) {
	$1 = AIM()
	$2 = 50 + (RANK() * 5)
	$10 = 8 + (RANK() / 4)
	while($10 > 0) {
		BULLET($1, $2)
		$10 -= 1
		$2 += 10
		WAIT(10)
	}
	WAIT(1000)
}
