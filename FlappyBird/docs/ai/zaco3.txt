// ザコ3 - 左右に動く。まっすぐな弾を撃つ
// ================================
// ログ有効
//LOG(1)
DECAY(93)
WAIT(2000)

// 左右に動く
REFLECT()
DECAY(100)
if(LOT()) {
	MOVE(90, 80)
}
else {
	MOVE(270, 80)
}
while(1) {
	$1 = 180
	$2 = 50 + (RANK() * 5)
	BULLET($1, $2)
	BULLET2(5, 0, $1, $2)
	BULLET2(10, 0, $1, $2)
	WAIT(1000)
}
