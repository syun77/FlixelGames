WAIT(1000)
DECAY(98)
// 3Way発射
$10 = 3
while($10 > 0) {
	$1 = AIM()
	$2 = 100
	BULLET($1-5, $2)
	BULLET($1, $2)
	BULLET($1+5, $2)
	$10 -= 1
	WAIT(1000)
}

$10 = 2
while($10 > 0) {
	MOVE2(100)
	$10 -= 1
	WAIT(1000)
}
